CREATE TABLE PLAYERS(P_ID INT NOT NULL,DOB VARCHAR2(25),HEIGHT VARCHAR2(10),SALARY DECIMAL(3,1), PRIMARY KEY(P_ID))
CREATE TABLE COACHES(C_ID INT NOT NULL,PRIMARY KEY(C_ID))
CREATE TABLE GENERAL_MANAGERS(GM_ID INT NOT NULL,PRIMARY KEY(GM_ID))
CREATE TABLE STAFF(S_ID INT NOT NULL,FIRSTNAME VARCHAR(50) NOT NULL,LASTNAME VARCHAR(50) NOT NULL,YEARS_OF_EXP INT,PRIMARY KEY(S_ID))
CREATE TABLE GAMES(G_ID INT NOT NULL,TEAM_ID INT NOT NULL,GAME_LOCATION VARCHAR(50),POINTS_FOR INT,POINTS_AGAINST INT,PRIMARY KEY(G_ID))
CREATE TABLE TEAMS(TEAM_ID INT  NOT NULL,TEAM_NAME VARCHAR(50) NOT NULL,WINS INT,LOSSES INT,"W/L%" DECIMAL(3,3),PRIMARY KEY(TEAM_ID))
CREATE TABLE LOCATIONS (PROVINCE VARCHAR(20) NOT NULL,CITY VARCHAR(20),PRIMARY KEY(PROVINCE))
CREATE TABLE TEAMS_MAPS_LOCATIONS(TEAM_ID INT NOT NULL,PROVINCE VARCHAR(20) NOT NULL,PRIMARY KEY(TEAM_ID, PROVINCE))
CREATE TABLE STATS(ST_ID INT NOT NULL,PPG DECIMAL(3,1),APG DECIMAL(3,1),RPG DECIMAL(3,1),TOV DECIMAL(3,1),STLS DECIMAL(3,1),BLKS DECIMAL(3,1),"FG%" DECIMAL(3,3),"FT%" DECIMAL(3,1),"3FG%" DECIMAL(3,1),PRIMARY KEY(ST_ID))
